Substitua estas funções no seu time-detalhe.html.

function extrairNomeTime(valor){
  if(typeof valor === "object" && valor){
    return normalizar(campo(valor,["nome","nomeTime","time","equipe"],""));
  }
  return normalizar(valor);
}

function jogosDoTime(){
  const nome = normalizar(nomeTime());

  return jogos.filter(j=>{
    const timeA = extrairNomeTime(campo(j,["timeA","mandante","timeMandante","nomeTimeA"],""));
    const timeB = extrairNomeTime(campo(j,["timeB","visitante","timeVisitante","nomeTimeB"],""));

    const okTime = timeA === nome || timeB === nome;

    const cat = campo(j,["categoria","categoriaNome"],"");
    const camp = campo(j,["campeonato","nomeCampeonato","campeonatoNome"],"");
    const campId = campo(j,["campeonatoId"],"");
    const ano = String(campo(j,["temporada","ano"], campo(j,["data","dataJogo"],"").slice(0,4)));

    const okCat = !filtroCategoria || cat === filtroCategoria;
    const okCamp = !filtroCampeonato || camp === filtroCampeonato || campId === filtroCampeonato;
    const okAno = !filtroAno || ano === filtroAno;

    return okTime && okCat && okCamp && okAno;
  });
}

function sumulasDoTime(){
  const nome = normalizar(nomeTime());

  return sumulas.filter(s=>{
    const timeA = extrairNomeTime(campo(s,["timeA","mandante","timeMandante","nomeTimeA"],""));
    const timeB = extrairNomeTime(campo(s,["timeB","visitante","timeVisitante","nomeTimeB"],""));

    const okTime = timeA === nome || timeB === nome;

    const cat = campo(s,["categoria","categoriaNome"],"");
    const camp = campo(s,["campeonato","nomeCampeonato","campeonatoNome"],"");
    const campId = campo(s,["campeonatoId"],"");
    const ano = String(campo(s,["temporada","ano"], campo(s,["data","dataJogo"],"").slice(0,4)));

    return okTime &&
      (!filtroCategoria || cat === filtroCategoria) &&
      (!filtroCampeonato || camp === filtroCampeonato || campId === filtroCampeonato) &&
      (!filtroAno || ano === filtroAno);
  });
}

function calcularStats(){
  let jogosQtd = 0, vitorias = 0, empates = 0, derrotas = 0, gp = 0, gc = 0;
  const nome = normalizar(nomeTime());

  jogosDoTime().forEach(j=>{
    jogosQtd++;

    const a = placarA(j);
    const b = placarB(j);

    const timeA = extrairNomeTime(campo(j,["timeA","mandante","timeMandante","nomeTimeA"],""));

    const ehA = timeA === nome;

    const golsPro = ehA ? a : b;
    const golsContra = ehA ? b : a;

    gp += golsPro;
    gc += golsContra;

    if(golsPro > golsContra) vitorias++;
    else if(golsPro === golsContra) empates++;
    else derrotas++;
  });

  return {
    jogos:jogosQtd,
    vitorias,
    empates,
    derrotas,
    gols:gp,
    golsContra:gc,
    saldo:gp-gc,
    pontos:vitorias*3+empates
  };
}
